            setTimeout(() => {
                const popup = document.createElement('div');

                popup.style.height = `${size}px`;
                popup.style.top = `${Math.max(0, yPos)}px`;
                popup.style.left = `${Math.max(0, xPos)}px`;

                popup.style.transform = 'translate(-50%, -50%) scale(1)';
